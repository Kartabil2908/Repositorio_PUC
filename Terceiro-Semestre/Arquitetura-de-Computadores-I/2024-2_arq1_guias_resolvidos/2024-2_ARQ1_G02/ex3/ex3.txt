a.) 0,011110(2)     = 0,132(4)

0,[01][11][10] = 0,132

---------------------------------------------

b.) 0,101001(2) = 0,51(8) 

0,[101][001] = 0,51(8)

----------------------------------------------------------------------------------------------------------------

c.) 0,100110(2)     = X(16)

0,[1001][1000] = 98
